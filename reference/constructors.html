Constructors

Constructors - used when you want to create multiple instances of your own objects

Constructor example
var Bicycle = function(color, numOfWheels) {
    this.color = color;
    this.numOfWheels = numOfWheels;
};

Creating a new instance of an object from the Constructor
var myBicycle = new Bicycle(blue, 2)

Accessing an object's properties
console.log(myBicycle.color) // blue

Adding a method to an object
Bicycle.prototype.speed(terrain) {
  if(terrain === street) {
    speed = 15;
   }
 else {
    speed = 10;
  }
};

myBicycle.speed(street); // speed = 15
